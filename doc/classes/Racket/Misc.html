randbits(size)

Return a number that is at most size bits long

# File lib/racket/misc/misc.rb, line 31
  def Misc.randbits(size)
    bits = 0
    srand Time.now.usec
    0.upto(size-1) {
      bits <<= 1
      bits |= rand(2)
    }
    bits
  end

  def Misc.randbytes(size)
    bytes = 0
    0.upto(size-1) {
      bytes <<= 8
      bytes |= randbits(8)
    }
    bytes
  end

  def Misc.randstring(size)
    s = ""
    0.upto(size-1) {
      s += sprintf("%c", randbytes(1))
    }
    s
  end

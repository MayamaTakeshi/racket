  def decode(data)
    s = "#{punpack_string(@ts)}#{punpack_string(@ls)}"
    type, length, tmp = data.unpack("#{s}a*")
    value, rest = tmp.unpack("a#{length}a*")
    [type, length, value, rest]
  end
